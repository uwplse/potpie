  unfold UIP_LogicPropAexpImp_Lang.LP.V in *;

Axiom UIP_AbsEnv :
  forall (l1 l2: AbsEnv)
    (p1 p2: l1 = l2),
    p1 = p2.

Lemma UIP_AbsEnv_refl :
  forall (l: AbsEnv)
    (p: l = l),
    p = eq_refl.
Proof.
  intros.
  pose proof (UIP_AbsEnv).
  specialize (H l l p (@eq_refl _ l)).
  eassumption.
Qed.
