Inductive Forall_set {A: Type}: (A -> Set) -> (list A) -> Type :=
| Forall_set_nil :
  forall (P: A -> Set),
    Forall_set P nil
| Forall_set_cons :
  forall (P: A -> Set) (a: A) (l: list A),
    P a ->
    Forall_set P l ->
    Forall_set P (cons a l).
